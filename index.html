// دالة لفحص الـ IP وكشف الـ VPN async function checkUserSecurity() { try { const response = await fetch('https://ipapi.co/json/'); const data = await response.json(); // إذا كان هناك مؤشر على VPN أو Proxy (يمكنك توسيع الفحص هنا) if (data.org.toLowerCase().includes("hosting") || data.org.toLowerCase().includes("vpn")) { alert("VPN/Proxy is not allowed!"); return false; } // تسجيل الـ IP في Firebase لمنع التكرار const userIp = data.ip; return userIp; } catch (error) { console.error("Security Check Failed", error); return false; } } // دالة تحديث Firebase عند نجاح التحقق function markAsVerified(userId, ip) { const db = firebase.database(); // فحص إذا كان هذا الـ IP مسجل لمستخدم آخر db.ref('used_ips/' + ip.replace(/\./g, '_')).once('value').then((snapshot) => { if (snapshot.exists() && snapshot.val() !== userId) { alert("This device/network is already linked to another account!"); } else { // تحديث حالة المستخدم في البوت db.ref('users/' + userId).update({ is_verified: true, ip: ip }); // تسجيل الـ IP في القائمة السوداء للمنع db.ref('used_ips/' + ip.replace(/\./g, '_')).set(userId); alert("Verification Successful! Return to the bot."); } }); }
